  \StrBefore[1]{#1}{-}[\myYear]%
  \StrBetween[1,2]{#1}{-}{-}[\myMonth]%
  \StrBehind[2]{#1}{-}[\myDay]%
  \printdate{\myDay}{\myMonth}{\myYear}%
}

      {4}{#3}%
